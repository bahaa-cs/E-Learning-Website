
CREATE TABLE IF NOT EXISTS `user_types` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `type` VARCHAR(150) NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE = InnoDB DEFAULT CHARACTER SET = utf8mb4;

CREATE TABLE IF NOT EXISTS `users` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `username` VARCHAR(150) NOT NULL,
  `password` VARCHAR(150) NOT NULL,
  `isBanned` TINYINT(1) NOT NULL,
  `user_type_id` INT(11) NOT NULL,
  PRIMARY KEY (`id`),
  INDEX `fk_users_user_types_idx` (`user_type_id`),
  CONSTRAINT `fk_users_user_types`
    FOREIGN KEY (`user_type_id`)
    REFERENCES `user_types` (`id`)
    ON DELETE CASCADE
    ON UPDATE CASCADE
) ENGINE = InnoDB DEFAULT CHARACTER SET = utf8mb4;


CREATE TABLE IF NOT EXISTS `courses` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `courseName` VARCHAR(150) NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE = InnoDB DEFAULT CHARACTER SET = utf8mb4;


CREATE TABLE IF NOT EXISTS `courses_streams` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `courses_id` INT(11) NOT NULL,
  PRIMARY KEY (`id`),
  INDEX `fk_courses_streams_courses1` (`courses_id`),
  CONSTRAINT `fk_courses_streams_courses1`
    FOREIGN KEY (`courses_id`)
    REFERENCES `courses` (`id`)
    ON DELETE CASCADE
    ON UPDATE CASCADE
) ENGINE = InnoDB DEFAULT CHARACTER SET = utf8mb4;


CREATE TABLE IF NOT EXISTS `streams_announcements` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `announcement` LONGTEXT NOT NULL,
  `courses_streams_id` INT(11) NOT NULL,
  `user_id` INT(11) NOT NULL,
  PRIMARY KEY (`id`),
  INDEX `fk_streams_announcements_courses_streams` (`courses_streams_id`),
  INDEX `fk_streams_announcements_users` (`user_id`),
  CONSTRAINT `fk_streams_announcements_courses_streams`
    FOREIGN KEY (`courses_streams_id`)
    REFERENCES `courses_streams` (`id`)
    ON DELETE CASCADE
    ON UPDATE CASCADE,
  CONSTRAINT `fk_streams_announcements_users`
    FOREIGN KEY (`user_id`)
    REFERENCES `users` (`id`)
    ON DELETE CASCADE
    ON UPDATE CASCADE
) ENGINE = InnoDB DEFAULT CHARACTER SET = utf8mb4;


CREATE TABLE IF NOT EXISTS `streams_assignments` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `assignment_desc` LONGTEXT NOT NULL,
  `courses_streams_id` INT(11) NOT NULL,
  `user_id` INT(11) NOT NULL,
  PRIMARY KEY (`id`),
  INDEX `fk_streams_assignments_courses_streams1_idx` (`courses_streams_id`),
  INDEX `fk_streams_assignments_users1_idx` (`user_id`),
  CONSTRAINT `fk_streams_assignments_courses_streams1`
    FOREIGN KEY (`courses_streams_id`)
    REFERENCES `courses_streams` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_streams_assignments_users1`
    FOREIGN KEY (`user_id`)
    REFERENCES `users` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION
) ENGINE = InnoDB DEFAULT CHARACTER SET = utf8mb4;


CREATE TABLE IF NOT EXISTS `streams_comments` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `comment` LONGTEXT NOT NULL,
  `type` ENUM('public', 'private') NOT NULL,
  `courses_streams_id` INT(11) NOT NULL,
  `user_id` INT(11) NOT NULL,
  PRIMARY KEY (`id`),
  INDEX `fk_streams_comments_courses_streams` (`courses_streams_id`),
  INDEX `fk_streams_comments_users` (`user_id`),
  CONSTRAINT `fk_streams_comments_courses_streams`
    FOREIGN KEY (`courses_streams_id`)
    REFERENCES `courses_streams` (`id`)
    ON DELETE CASCADE
    ON UPDATE CASCADE,
  CONSTRAINT `fk_streams_comments_users`
    FOREIGN KEY (`user_id`)
    REFERENCES `users` (`id`)
    ON DELETE CASCADE
    ON UPDATE CASCADE
) ENGINE = InnoDB DEFAULT CHARACTER SET = utf8mb4;


CREATE TABLE IF NOT EXISTS `submitted_assignments` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `assignment` BLOB NOT NULL,
  `streams_assignments_id` INT(11) NOT NULL,
  `user_id` INT(11) NOT NULL,
  PRIMARY KEY (`id`),
  INDEX `fk_submitted_assignments_streams_assignments1_idx` (`streams_assignments_id`),
  INDEX `fk_submitted_assignments_users1_idx` (`user_id`),
  CONSTRAINT `fk_submitted_assignments_streams_assignments1`
    FOREIGN KEY (`streams_assignments_id`)
    REFERENCES `streams_assignments` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_submitted_assignments_users1`
    FOREIGN KEY (`user_id`)
    REFERENCES `users` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION
) ENGINE = InnoDB DEFAULT CHARACTER SET = utf8mb4;
